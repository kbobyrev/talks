\begin{frame}
\frametitle{Algorithm description}
\begin{algorithmic}
  \STATE Initialize $S_0 \neq {terminal}$
  \STATE Select $A_0$ according to $\pi(. | S_0)$
  \STATE Store $S_0$, $A_0$, $Q(S_0, A_0)$
  \FOR{$t = 0, \ldots, T + n - 1$}
    \IF{$t < T$}
      \STATE Take Action $A_t$, observe $R$ and store $S_{t + 1}$
    \ENDIF
  \ENDFOR
\end{algorithmic}
\end{frame}
